\[
\begin{align*}
\cdot \text{Elementare Ausgüsse} &: 1, 2, 3, 4, 5, 6 \\
\cdot \text{Ergänzssmenge} \quad \Omega = \{1, 2, 3, 4, 5, 6\} \\
\cdot \text{Würfel eine gewohn Zahl} : A = \{2, 4, 6\} \\
&\quad A \subseteq \Omega \\
\cdot \text{Würfel eine ungewohn Zahl} : B = \{1, 3, 5\} \\
\cdot \text{A und B sind disjunkt} : A \cap B = \emptyset
\end{align*}
