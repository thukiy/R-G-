<|ref|>equation<|/ref|><|det|>[[137, 205, 546, 250]]<|/det|>
\[ \bullet \quad A = \{\langle K, K \rangle, \langle K, Z \rangle\} \quad \text{mit } P(A) = \frac{|A|}{|Z|} = \frac{2}{4} = \frac{1}{2} \]

<|ref|>equation<|/ref|><|det|>[[137, 258, 544, 302]]<|/det|>
\[ \bullet \quad B = \{\langle K, K \rangle, \langle Z, K \rangle\} \quad \text{mit } P(B) = \frac{|B|}{|Z|} = \frac{2}{4} = \frac{1}{2}\]

<|ref|>equation<|/ref|><|det|>[[137, 308, 544, 353]]<|/det|>
\[ \bullet \quad C = \{\langle K, K \rangle, \langle Z, Z \rangle\} \quad \text{mit } P(C) = \frac{|C|}{|Z|} = \frac{2}{4} = \frac{1}{2}\]
 

<|ref|>equation<|/ref|><|det|>[[137, 358, 565, 402]]<|/det|>
\[ \bullet \quad A \cap B = \{\langle K, K \rangle\} \quad \text{mit } P(A \cap B) = \frac{|A \cap B|}{|Z|} = \frac{1}{4} \]

<|ref|>equation<|/ref|><|det|>[[137, 408, 565, 453]]<|/det|>
\[ \bullet \quad A \cap C = \{\langle K, K \rangle\} \quad \text{mit } P(A\cap C) = \frac{|A \cap C|}{|Z|} = \frac{1}{4} \]

<|ref|>equation<|/ref|><|det|>[[0, 0, 0, 0]]<|/det|>
\[ \bullet \quad B \cap C = \{\langle K, K \rangle\} \quad \text{mit} P(B \cap C) = \frac{|B \cap C|}{|Z|} = \frac{1}{4} \]

<|ref|>equation
\[ \bullet \quad A \cap B \cap C = \{\langle K, K \rangle\} \quad \text{ mit } P(A \cap B \cap C) = \frac{|A \cap B \cap C|}{|Z|} = \frac{1}{4} \]

<|ref|>equation

<|ref|>equation

<|ref|>equation
