b^{\prime \prime}(x) & =6 \cdot 2 x^{2-1}+6-0=12 x+6
\end{aligned}
$$

$$
\begin{array}{ll}

\end{array}

$$
x_{1}=-\frac{1}{2}-\frac{3}{2}=-\frac{4}{2}=-2 \quad \text { und } \quad x_{2}=-\frac{1}{2}+\frac{3}{2}=\frac{2}{2}=1
