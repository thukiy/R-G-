<|ref|>equation<|/ref|><|det|>[[120, 668, 601, 780]]<|/det|>
\[\begin{align*}
· z &= 2e^{i\frac{\pi}{6}} \quad r &= 2, \quad \varphi = \frac{\pi}{6} \\
&= 2(\cos\frac{\pi}{6} + i \cdot \sin\frac{\pi}{6}) \\
&= 2\left(\frac{\sqrt{3}}{2} + i \cdot \frac{1}{2}\right) = \sqrt{3} + i
\end{align*}\]

<|ref|>equation<|/ref|><|det|>[[120, 806, 550, 952]]<|/det|>
\[\begin{align*}
· z_1 &= 2e^{i\frac{\pi}{6}} \quad z_2 &= 3e^{i\frac{\pi}{4}\pi} \\
&= 2 \cdot 2e^{i\frac{\pi}{6}} e^{i\frac{\pi}{4}\pi} \\
&= 6 \cdot e^{i\left(\frac{\pi}{6} + \frac{\pi}{4}\pi\right)} = 6 \cdot e^{i\frac{3\pi}{2}} \\
&= 6 \cdot (\cos\left(\frac{3\pi}{2}\right) + i \cdot \sin\left(\frac{3\pi}{2}\right)) = -6i
\end{align*}\]
