<|ref|>equation<|/ref|><|det|>[[186, 51, 646, 120]]<|/det|>
\[
\begin{align*}
n_{1,2} &= -8 \pm i \sqrt{w_0^2 - 8^2} \\
&= -8 \pm i w_d \\
&= w_d
\end{align*}
\]

<|ref|>text<|/ref|><|det|>[[186, 125, 850, 234]]<|/det|>

\(= e^{-8(t-t_0)} [A \sin(w_d(t-t_0)) + B \cdot \cos(w_d(t-t_0))]\)
