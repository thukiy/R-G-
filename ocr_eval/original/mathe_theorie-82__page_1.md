## Musaelechnung 

<|ref|>equation<|/ref|><|det|>[[50, 278, 583, 360]]<|/det|>
  \[  \begin{aligned}
  a &= \sqrt[n]{b} \iff a^n = b \\
  \text{Radikand} &
  \begin{cases}
  a, b \in \mathbb{R}^+ \\
  n \ge 1
  \end{cases}
  \end{aligned}\]

<|ref|>equation<|/ref|><|det|>[[50, 377, 199, 408]]<|/det|>
  \[  \frac{n}{\sqrt[n]{b}} = b^{\frac{n}{n}}\]

<|ref|>equation<|/ref|><|det|>[[50, 420, 201, 451]]<|/det|>
  \[  \frac{n}{\sqrt[n]{a \cdot b}} = \frac{n}{\sqrt[n]{a}} \cdot \frac{n}{\sqrt[n]{b}}\]

<|ref|>equation<|/ref|><|det|>[[50, 454, 300, 484]]<|/det|>
  \[  \frac{n}{\sqrt[n]{a+b}} \neq \frac{n}{\sqrt[n]{a}} + \frac{n}{\sqrt[n]{b}}\]
