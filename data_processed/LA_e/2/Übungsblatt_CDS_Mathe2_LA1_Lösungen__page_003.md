<|ref|>equation<|/ref|><|det|>[[114, 123, 748, 220]]<|/det|>
\[
\begin{align*}
\frac{(z_1 \cdot z_2)^*}{z_1 \cdot z_2} &= \left( (x_1 + y_1i) \cdot (x_2 + y_2i) \right)^* = (x_1x_2 - y_1y_2 + x_1y_2i + x_2y_1i)^* \\
&= \left( x_1x_2 - y_1y_2 + (x_1y_2 + x_2y_1) \cdot i \right)^* = x_1x_2 - y_1y_2 - (x_1y_2 + x_2y_1) \cdot i \\
&= x_1x_2 - y_1y_2 - x_1y_2i - x_2y_1i = (x_1 - y_1i) \cdot (x_2 - y_2i) = z_1^* \cdot z_2^*.
\end{align*}
\]

d) 

<|ref|>equation<|/ref|><|det|>[[120, 238, 590, 275]]<|/det|>

\left( \frac{z_1}{z_2} \right)^* = \left( \frac{x_1 + y_1i}{x_2 + y_2i} \right)^* = \left( \frac{x_1x_2 + y_1y_2}{x_2^2 + y_2^2} + \frac{x_2y_1 - x_1y_2}{x_2^2 + y_2^2} \cdot i \right)^*
\]

<|ref|>equation<|/ref|><|det|>[[175, 290, 794, 370]]<|/det|>

& = \frac{x_1x_2 + y_1y_2}{x_2 + y_2^2} - \frac{x_2y_1 - x_1y_2}{x_2 + y_2^2} \cdot i = \frac{x_1x_2 + y_1y_2}{x_1^2 + y_2^2} + \frac{-x_2y_1 + x_1y_2}{x_2^2 + y_2^2} - i = \frac{x_1 - y_1i}{x_2 - y_2i} \\
& = \frac{z_1^*}{z_2^*}.
\end{align*}
\]
