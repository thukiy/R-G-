<|ref|>equation<|/ref|><|det|>[[100, 415, 888, 620]]<|/det|>
\[ \begin{array}{l@{\quad}l} \cdot f_0(p) & f \\ \cdot & \vdots \\ \cdot & f_n(p) \\ \cdot & \vdots \\ \cdot & f_0(q) \\ \cdot & \vdots \\ \cdot & f_0(q) \end{array} \quad \begin{array}{l@{\quad}l} \lim_{n \to \infty} f_n(x) = f(x) \\ \text{Folge } (f_n(p)) \to f(p) : \\ f_0(p), f_1(p), f_2(p) \dots \to f(p) \end{array} \]
