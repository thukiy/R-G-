c) Wir betrachten die Funktion

$$
f(x)=\left(x^{3}-1\right) \cdot(1-x)
$$

F端r die Ableitung erhalten wir

$$
\begin{aligned}
\underline{\underline{f^{\prime}(x)}} & =\left(x^{3}-1\right)^{\prime} \cdot(1-x)+\left(x^{3}-1\right) \cdot(1-x)^{\prime}=3 x^{2} \cdot(1-x)+\left(x^{3}-1\right) \cdot(-1) \\
& =3 x^{2}-3 x^{3}+1-x^{3}=\underline{\underline{3 x^{2}-4 x^{3}+1}}
\end{aligned}
$$

d) Wir betrachten die Funktion

$$
f(x)=\left(x^{2}-x+1\right)^{2}=\left(x^{2}-x+1\right) \cdot\left(x^{2}-x+1\right)
$$

F端r die Ableitung erhalten wir

$$
\begin{aligned}
\underline{\underline{f^{\prime}(x)}} & =\left(x^{2}-x+1\right)^{\prime} \cdot\left(x^{2}-x+1\right)+\left(x^{2}-x+1\right) \cdot\left(x^{2}-x+1\right)^{\prime} \\
& =2\left(x^{2}-x+1\right) \cdot\left(x^{2}-x+1\right)^{\prime}=2\left(x^{2}-x+1\right) \cdot(2 x-1) \\
& =4 x^{3}-4 x^{2}+4 x-2 x^{2}+2 x-2=\underline{\underline{4 x^{3}-6 x^{2}+6 x-2}}
\end{aligned}
$$

e) Wir betrachten die Funktion

$$
f(x)=(x+1) \cdot \sqrt{x}
$$

F端r die Ableitung erhalten wir

$$
\begin{aligned}
\underline{\underline{f^{\prime}(x)}} & =(x+1)^{\prime} \cdot \sqrt{x}+(x+1) \cdot(\sqrt{x})^{\prime}=(1+0) \cdot \sqrt{x}+(x+1) \cdot \frac{1}{2 \cdot \sqrt{x}}=\sqrt{x}+\frac{x+1}{2 \cdot \sqrt{x}} \\
& =\frac{2 \cdot \sqrt{x} \cdot \sqrt{x}}{2 \cdot \sqrt{x}}+\frac{x+1}{2 \cdot \sqrt{x}}=\frac{2 \cdot x+x+1}{2 \cdot \sqrt{x}}=\underline{\underline{\frac{3 x+1}{2 \sqrt{x}}}}
\end{aligned}
$$

f) Wir betrachten die Funktion

$$
f(x)=(x+1) \cdot \frac{1}{x}
$$

F端r die Ableitung erhalten wir

$$
\begin{aligned}
\underline{\underline{f^{\prime}(x)}} & =(x+1)^{\prime} \cdot \frac{1}{x}+(x+1) \cdot\left(\frac{1}{x}\right)^{\prime}=(1+0) \cdot \frac{1}{x}+(x+1) \cdot(-1) \cdot \frac{1}{x^{2}}=\frac{1}{x}-\frac{x+1}{x^{2}} \\
& =\frac{x}{x^{2}}-\frac{x+1}{x^{2}}=\frac{x-x-1}{x^{2}}=\frac{0-1}{x^{2}}=\underline{\underline{-\frac{1}{x^{2}}}}
\end{aligned}
$$
