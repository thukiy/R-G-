\[ f(x) : y = 1 \cdot x + a \]

<|ref|>equation<|/ref|><|det|>[[90, 220, 590, 327]]<|/det|>
\[ x_s = \frac{1}{4a^2} \int_0^{2a} f(x \cdot (x+a)) \, dx = \frac{1}{4a^2} \int_0^{2a} (x^2 + ax) \, dx \]
\[ = \frac{1}{4a^2} \left[ \frac{1}{3} x^3 + \frac{9}{2} x^2 \right]_0^{2a} = \frac{1}{4a^2} \left( \frac{1}{3} \cdot 8a^3 + 2a^3 \right) \]
\[ = \frac{1}{4} \cdot \frac{14}{3} \cdot \frac{1}{a^2} \cdot a^3 = \frac{7}{6} a \]

<|ref|>equation<|/ref|><|det|>[[90, 352, 710, 460]]<|/det|>
\[ y_s = \frac{1}{4a^2} \int_0^{2a}\left(\frac{1}{2}(x+a)^2\right)dx = \frac{1}{4a^2} \int_0^{2a}(x^2 + 2ax + a^2)dx \]
\[ = \frac{1}{4a^2} \left[ \frac {1}{3} x^3 + ax^2 + a^2 x \right]_0^{2a} = \frac{1}{4a^2} ( \frac{1}{3} \cdot 8a^3 + 4a^3 + 2a^3 ) \]
\[ = \frac{26}{24} a = \frac{13}{12} a \]
